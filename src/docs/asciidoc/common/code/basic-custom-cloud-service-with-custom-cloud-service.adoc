.Basic custom cloud service with other custom cloud service
[source, javascript, linenums]
----
const { Inject } = require('@zetapush/platform');

class DateCustomService { <1>

    static get parameters {
        return [];
    }

    constructor() {}

    formatDate(date) {
        return date.toLocalDateString("fr-FR");
    }
}

class HelloWorldService {

    static get parameters {
        return [
            new Inject(DateCustomService)
        ];
    }

    constructor(dateCustomService) { <2>
        this.dateService = dateCustomService;
    }

    helloWorldWithDate() {
        const date = this.dateService.formateDate(new Date()); <3>
        return `Hello world at ${date}`;
    }
}
----
<1> First _custom cloud service_ only to manipulate dates
<2> `dateCustomService` is an instance of `DateCustomService`
<3> We use the `DateCustomService` in our `HelloWorldService`
